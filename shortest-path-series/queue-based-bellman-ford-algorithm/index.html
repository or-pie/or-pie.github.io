队列版 Bellman-Ford 算法

在 Bellman-Ford 算法中，并不是所有的松弛操作都是必要的。

1. 队列版 Bellman-Ford 算法

回顾 Bellman-Ford 算法对边 \((u, v)\) 的松弛操作，其本质是利用 \(dist[u]\) 来尝试降低 \(dist[v]\)。标准 Bellman-Ford 算法的每一轮都会对所有边进行松弛操作，但是并不是所有松弛操作都是必要的。考虑某条边 \((u, v)\)，如果在某个时刻已经满足 \(dist[v] \le dist[u] + w(u, v)\)，那么只要 \(dist[u]\) 不发生改变，那么该不等式总是成立的，没有必要反复对边 \((u, v)\) 进行松弛操作。显然，只有 \(dist\) 值发生改变的节点发出的边才需要进行松弛操作。

我们用一个 队列 来维护 “哪些节点可能会引起有效的松弛操作”，并对 Bellman-Ford 算法进行基于队列的优化改进。改进的算法流程类似于广度优先搜索，但每个节点可能会进出队列多次。算法不断从队列中取出一个节点 \(u\)，对它发出的每条边 \((u, v) \in A[u]\) 进行松弛操作，如果松弛操作成功降低了 \(dist[v]\)，那么就把节点 \(v\) 加入队列。这里，我们保证每个节点在任意时刻只会在队列里出现一次。

那么，怎么检测负权重环呢？首先，如果在某次松弛操作后发现 \(dist[s] < 0\)，那么算法可以立刻返回 false，因为算法检测到一个包含源点 \(s\) 的负权重环。其次，我们为每个节点 \(v \in V\) 添加属性 \(rank[v]\)，记录它在从源点 \(s\) 到 \(v\) 的当前已知最短路径上的序号。特殊地，令 \(rank[s] = 0\)。如果在一次松弛操作后发现 \(rank[v] = |V|\)，说明从 \(s\) 到 \(v\) 的当前已知最短路径包含 \(|V|\) 条边，必然有节点被访问了至少两次，因此可断定输入图中含有从源点可达的负权重环。

bool relax(int u, int v, double w) {
  if (dist[v] > dist[u] + w) {
    dist[v] = dist[u] + w;
    prev[v] = u;
    rank[v] = rank[u] + 1;
    return true;
  } else {
    return false;
  }
}

bool bellman_ford_queue(const Graph &G, int s) {
  dist.assign(G.n, inf);
  prev.assign(G.n, nil);
  rank.assign(G.n, nil);
  dist[s] = 0;
  rank[s] = 0;
  Queue Q(G.n);
  Q.push(s);
  while (not Q.empty()) {
    int u = Q.pop();
    for (auto [v, w] : G.A[u]) {
      if (relax(u, v, w)) {
        if (dist[s] < 0 || rank[v] == G.n) {
          clue = v;
          return false;
        }
        if (not Q.contains(v)) {
          Q.push(v);
        }
      }
    }
  }
  clue = nil;
  return true;
}

在多数情况下队列版 Bellman-Ford 算法效率比标准版更高（特别是在稀疏图上），但其最坏情况下的运行时间依然是 \(O(|V| |E|)\)。

#include "bellman-ford-queue.h"

int main() {
  const char *names = "stxyz";
  Graph G(5);
  G.add_edge(0, 1, 6);
  G.add_edge(0, 3, 7);
  G.add_edge(1, 2, 5);
  G.add_edge(1, 3, 8);
  G.add_edge(1, 4, -4);
  G.add_edge(2, 1, -2);
  G.add_edge(3, 2, -3);
  G.add_edge(3, 4, 9);
  G.add_edge(4, 0, 2);
  G.add_edge(4, 2, 7);

  BellmanFordQueue BFQ;
  if (BFQ.bellman_ford_queue(G, 0)) {
    std::cout << "从源点 " << names[0] << " 出发的最短路径：" << std::endl;
    for (int v = 0; v < G.n; v++) {
      std::vector<int> p = BFQ.reconstruct_shortest_path(v);
      std::cout << names[v] << " (" << BFQ.dist[v] << "): ";
      for (int i = 0; i < p.size(); i++) {
        std::cout << names[p[i]] << (i < p.size() - 1 ? "->" : "");
      }
      std::cout << std::endl;
    }
  } else {
    std::cout << "存在从源点 " << names[0] << " 可达的负权重环：";
    std::vector<int> c = BFQ.reconstruct_negative_weight_cycle();
    for (int i = 0; i < c.size(); i++) {
      std::cout << names[c[i]] << (i < c.size() - 1 ? "->" : "");
    }
    std::cout << std::endl;
  }

  return 0;
}

#ifndef BELLMAN_FORD_QUEUE_H
#define BELLMAN_FORD_QUEUE_H

#include "graph.h"
#include "queue.h"

class BellmanFordQueue {
public:
  std::vector<double> dist;
  std::vector<int> prev;
  std::vector<int> rank;
  int clue;

  bool relax(int u, int v, double w) {
    if (dist[v] > dist[u] + w) {
      dist[v] = dist[u] + w;
      prev[v] = u;
      rank[v] = rank[u] + 1;
      return true;
    } else {
      return false;
    }
  }

  bool bellman_ford_queue(const Graph &G, int s) {
    dist.assign(G.n, inf);
    prev.assign(G.n, nil);
    rank.assign(G.n, nil);
    dist[s] = 0;
    rank[s] = 0;
    Queue Q(G.n);
    Q.push(s);
    while (not Q.empty()) {
      int u = Q.pop();
      for (auto [v, w] : G.A[u]) {
        if (relax(u, v, w)) {
          if (dist[s] < 0 || rank[v] == G.n) {
            clue = v;
            return false;
          }
          if (not Q.contains(v)) {
            Q.push(v);
          }
        }
      }
    }
    clue = nil;
    return true;
  }

  std::vector<int> reconstruct_shortest_path(int v) {
    if (clue != nil || dist[v] == inf) {
      return {};
    } else {
      std::vector<int> p;
      for (int x = v; x != nil; x = prev[x]) {
        p.push_back(x);
      }
      std::reverse(p.begin(), p.end());
      return p;
    }
  }

  std::vector<int> reconstruct_negative_weight_cycle() {
    if (clue == nil) {
      return {};
    } else {
      std::vector<bool> aux(prev.size(), false);
      std::vector<int> p;
      int x = clue;
      while (not aux[x]) {
        aux[x] = true;
        p.push_back(x);
        x = prev[x];
      }
      std::vector<int> c = {x};
      for (auto it = p.rbegin(); *it != x; it++) {
        c.push_back(*it);
      }
      c.push_back(x);
      return c;
    }
  }
};

#endif

#ifndef QUEUE_H
#define QUEUE_H

#include <bits/stdc++.h>

class Queue {
private:
  std::deque<int> data;
  std::vector<bool> in_queue;

public:
  Queue(int n) : data(), in_queue(n, false) {}

  bool empty() { return data.empty(); }

  bool contains(int v) { return in_queue[v]; }

  void push(int v) {
    data.push_back(v);
    in_queue[v] = true;
  }

  int pop() {
    int v = data.front();
    data.pop_front();
    in_queue[v] = false;
    return v;
  }
};
#endif

#ifndef GRAPH_H
#define GRAPH_H

#include <bits/stdc++.h>

class Graph {
public:
  int n;
  std::vector<std::tuple<int, int, double>> E;
  std::vector<std::vector<std::pair<int, double>>> A;

  Graph(int n) : n(n), E(), A(n) {}

  void add_edge(int u, int v, double w) {
    E.emplace_back(u, v, w);
    A[u].emplace_back(v, w);
  }

  void clear() {
    E.clear();
    A.assign(n, {});
  }
};

const double inf = std::numeric_limits<double>::infinity();
const int nil = -1;

3. 其他优化技巧

Bellman-Ford 算法的实际运行效率很大程度上取决于松弛操作的顺序，上面介绍的改进算法利用先进先出的队列来管理待处理节点。事实上，节点可以不遵循先进先出的顺序进行处理。有的时候，启发式地调整节点出队的顺序（可能）可以进一步提高算法的平均性能（但仍无法提高最坏情况下的性能）。

最著名的两种技巧在将节点压入队列 \(Q\) 时，通过重新调整队列中元素的顺序，从而使得更靠近源点的节点能够被更早地处理。因此一旦实现了这两种技巧，\(Q\) 不再是一个严格的先进先出队列，而更像一个链表或双端队列。

  • 第一个技巧是 距离小者优先（Small Label First, SLF）：如果 \(dist[v]\) 值小于当前队头元素的 \(dist\) 值，那么将节点 \(v\) 压入队头，否则将 \(v\) 压入队头；
  • 第二个技巧是 距离大者置后（Large Label Last, LLL）：把节点 \(v\) 压入队尾后，计算队列中的平均 \(dist\) 值，按以下规则不断调整队列：只要队头元素的 \(dist\) 值大于平均 \(dist\) 值，就将其取出重新压入队尾，直到队头元素的 \(dist\) 值小于或等于平均 \(dist\) 值；

我们也可以考虑使用 优先队列 来管理待处理节点：算法不断从队列中取出 \(dist\) 值最小的节点，以保证每一次都处理最靠近源点的待处理节点。使用优先队列的 Bellman-Ford 算法也可以看作是 Dijkstra 算法在含负权重边的最短路径问题上的扩展。
