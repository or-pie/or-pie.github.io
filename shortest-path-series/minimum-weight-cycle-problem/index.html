最小权重环问题

最小权重环问题可以使用全节点对最短路径算法（如 Floyd-Warshall 算法、Johnson 算法）解决。

1. 问题定义

给定有向图 \(G = (V, E)\) 和权重函数 \(w : E \rightarrow \mathbb{R}\)，最小权重环问题 旨在找到输入图中具有最小权重的环，其权重称为 最小环权重。如果输入图中存在负权重环，那么最小环权重为 \(-\infty\)，最小权重环则没有明确定义。

2. 算法思想

最小权重环问题可以使用全节点对最短路径算法解决，例如 Floyd-Warshall 算法和 Johnson 算法。解决方法依赖于下面的观察。

最小权重环一定包含最短路径

设环 \(c = \langle v_0, v_1, \dots, v_k \rangle\) 为包含边 \((v_0, v_1)\) 的最小权重环，其中 \(v_0 = v_k\)，那么路径 \(p = \langle v_1, \dots, v_k\rangle\) 是从 \(v_1\) 到 \(v_k\) 的一条最短路径。

证明

如果把环 \(c\) 分解为 \(\langle v_0 \rightarrow v_1 \stackrel{p}{\rightsquigarrow} v_k \rangle\)。现在，假设存在一条从 \(v_1\) 到 \(v_k\) 的路径 \(p'\)，其权重 \(w(p') < w(p)\)，则环 \(c' = \langle v_0 \rightarrow v_1 \stackrel{p'}{\rightsquigarrow} v_k \rangle\) 的权重 \(w(c') < w(c)\)，这与 \(c\) 是最小权重环这一前提矛盾。

首先在输入图上运行一次全节点对最短路径算法，例如 Floyd-Warshall 算法或 Johnson 算法。如果输入图中不存在负权重环，对于任意边 \((u, v) \in E\)，设 \(p\) 为从节点 \(v\) 到节点 \(u\) 的一条最短路径，即 \(w(p) = \delta(v, u)\)，那么 \(\langle u \rightarrow v \stackrel{p}{\rightsquigarrow} u\rangle\) 就是包含边 \((u, v)\) 的最小权重环，它从节点 \(u\) 出发，经过边 \((u, v)\) 到达 \(v\)，再路径 \(p\) 回到节点 \(u\)。枚举图中所有的边，即可得到图 \(G\) 的最小权重环，对应的最小环权重为 \(\min_{(u, v) \in E} (w(u, v) + \delta(v, u))\)。

下面的代码描述了在 Floyd-Warshall 算法运行结束以后用于重构最小权重环的算法。注意，在重构最小权重环时，我们使用空路径表示输入图包含负权重环（第 3 行）或者输入图不包含任何环（第 14 行）。

std::vector<int> reconstruct_minimum_weight_cycle(const Graph &G) {
  if (FW.clue != nil) {
    return {};
  } else {
    double d = inf;
    std::pair<int, int> e;
    for (auto [u, v, w] : G.E) {
      if (FW.dist[v][u] != inf && d > w + FW.dist[v][u]) {
        d = w + FW.dist[v][u];
        e = {u, v};
      }
    }
    if (d == inf) {
      return {};
    } else {
      auto [u, v] = e;
      std::vector<int> c = {u};
      for (int x = u; FW.prev[v][x] != nil; x = FW.prev[v][x]) {
        c.push_back(FW.prev[v][x]);
      }
      c.push_back(u);
      std::reverse(c.begin(), c.end());
      return c;
    }
  }
}

在代码中，FW 表示 Floyd-Warshall 算法对象。

#include "minimum-weight-cycle.h"

int main() {
  Graph G(5);
  G.add_edge(0, 1, 3);
  G.add_edge(0, 2, 8);
  G.add_edge(0, 4, -4);
  G.add_edge(1, 3, 1);
  G.add_edge(1, 4, 7);
  G.add_edge(2, 1, 4);
  G.add_edge(3, 0, 2);
  G.add_edge(3, 2, -5);
  G.add_edge(4, 3, 6);

  MinimumWeightCycle MWC;
  if (MWC.floyd_warshall(G)) {
    std::vector<int> c = MWC.reconstruct_minimum_weight_cycle(G);
    if (c.empty()) {
      std::cout << "输入图不包含任何环！" << std::endl;
    } else {
      std::cout << "最小权重环：";
      for (int i = 0; i < c.size(); i++) {
        std::cout << c[i] << (i < c.size() - 1 ? "->" : "");
      }
      std::cout << std::endl;
    }
  } else {
    std::cout << "存在负权重环：";
    std::vector<int> c = MWC.reconstruct_negative_weight_cycle();
    for (int i = 0; i < c.size(); i++) {
      std::cout << c[i] << (i < c.size() - 1 ? "->" : "");
    }
    std::cout << std::endl;
  }

  return 0;
}

class MinimumWeightCycle {
public:
  FloydWarshall FW;

  bool floyd_warshall(const Graph &G) { return FW.floyd_warshall(G); }

  std::vector<int> reconstruct_minimum_weight_cycle(const Graph &G) {
    if (FW.clue != nil) {
      return {};

      double d = inf;
      std::pair<int, int> e;

    }
  }

  std::vector<int> reconstruct_negative_weight_cycle() {
    return FW.reconstruct_negative_weight_cycle();
  }
};

#ifndef FLOYD_WARSHALL_H
#define FLOYD_WARSHALL_H

#include "graph.h"

class FloydWarshall {
public:
  std::vector<std::vector<double>> dist;
  std::vector<std::vector<int>> prev;
  int clue;

  bool floyd_warshall(const Graph &G) {
    dist.assign(G.n, std::vector<double>(G.n, inf));
    prev.assign(G.n, std::vector<int>(G.n, nil));
    for (int i = 0; i < G.n; i++) {
      dist[i][i] = 0;
    }
    for (auto [i, j, w] : G.E) {
      if (dist[i][j] > w) {
        dist[i][j] = w;
        prev[i][j] = i;
        if (dist[i][i] < 0) {
          clue = i;
          return false;
        }
      }
    }
    for (int k = 0; k < G.n; k++) {
      for (int i = 0; i < G.n; i++) {
        for (int j = 0; j < G.n; j++) {
          if (dist[i][j] > dist[i][k] + dist[k][j]) {
            dist[i][j] = dist[i][k] + dist[k][j];
            prev[i][j] = prev[k][j];
            if (dist[i][i] < 0) {
              clue = i;
              return false;
            }
          }
        }
      }
    }
    clue = nil;
    return true;
  }

  std::vector<int> reconstruct_shortest_path(int u, int v) {
    if (clue != nil || dist[u][v] == inf) {
      return {};
    } else {
      std::vector<int> p;
      for (int x = v; x != nil; x = prev[u][x]) {
        p.push_back(x);
      }
      std::reverse(p.begin(), p.end());
      return p;
    }
  }

  std::vector<int> reconstruct_negative_weight_cycle() {
    if (clue == nil) {
      return {};
    } else {
      std::vector<bool> aux(prev.size(), false);
      std::vector<int> p;
      int x = clue;
      while (not aux[x]) {
        aux[x] = true;
        p.push_back(x);
        x = prev[clue][x];
      }
      std::vector<int> c = {x};
      for (auto it = p.rbegin(); *it != x; it++) {
        c.push_back(*it);
      }
      c.push_back(x);
      return c;
    }
  }
};

#endif

#ifndef GRAPH_H
#define GRAPH_H

#include <bits/stdc++.h>

class Graph {
public:
  int n;
  std::vector<std::tuple<int, int, double>> E;
  std::vector<std::vector<std::pair<int, double>>> A;

  Graph(int n) : n(n), E(), A(n) {}

  void add_edge(int u, int v, double w) {
    E.emplace_back(u, v, w);
    A[u].emplace_back(v, w);
  }

  void clear() {
    E.clear();
    A.assign(n, {});
  }
};

const double inf = std::numeric_limits<double>::infinity();
const int nil = -1;

#endif
