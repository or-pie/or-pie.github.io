懒惰 Dijkstra 算法

虽然大多数编程语言的标准库提供了基于二叉堆的优先队列数据结构，但它们通常不支持 decrease_key 操作，使标准 Dijkstra 算法的实现变得十分繁琐。

1. 懒惰 Dijkstra 算法

在上一篇文章中我们总结道：当 \(|E|= O(|V|^2 \mathbin{/} \log |V|)\) 时，基于二叉堆的 Dijkstra 算法性能更好。事实上，如果没有对图的稠密性做特殊说明，我们通常认为基于二叉堆的 Dijkstra 算法就是 标准 Dijkstra 算法。然而，虽然大多数编程语言的标准库提供了基于二叉堆的优先队列数据结构，但它们通常不支持 decrease_key 操作，使标准 Dijkstra 算法的实现变得十分繁琐。

幸运的是，我们依然可以使用不提供 decrease_key 操作的优先队列来实现 Dijkstra 算法，并且保持与标准 Dijkstra 算法相当的运行速度，只需要对算法做如下改变：

  • 当成功降低了某个已经在优先队列中的节点的最短路径估计值，我们不使用 decrease_key 操作，而是简单地使用 insert 操作将带有新最短路径估计值的节点插入到优先队列中；
  • 从优先队列中取出节点后，如果这不是该节点第一次被取出，我们会直接忽略它。

我们称这个版本的 Dijkstra 算法为 懒惰 Dijkstra 算法，因为我们允许节点重复进入优先队列，并且延迟了对重复节点的删除操作。

bool relax(int u, int v, double w) {
  if (dist[v] > dist[u] + w) {
    dist[v] = dist[u] + w;
    prev[v] = u;
    return true;
  } else {
    return false;
  }
}

void lazy_dijkstra(const Graph &G, int s) {
  dist.assign(G.n, inf);
  prev.assign(G.n, nil);
  dist[s] = 0;
  PriorityQueue Q;
  Q.insert(s, 0);
  while (not Q.empty()) {
    auto [u, du] = Q.extract_min();
    if (dist[u] == du) {
      for (auto [v, w] : G.A[u]) {
        if (relax(u, v, w)) {
          Q.insert(v, dist[v]);
        }
      }
    }
  }
}

在懒惰 Dijkstra 算法中，由于节点允许重复进入优先队列，优先队列的元素数量从 \(O(|V|)\) 增加到 \(O(|E|)\)。因此，算法执行了 \(O(|E|)\) 次 extract_min 和 insert 操作，总运行时间是 \(O(|E| \log |E|)\)。考虑到 \(|E| = O(|V|^2)\)，这个运行时间也可以表示为 \(O(|E| \log |V|)\)，即懒惰 Dijkstra 算法和标准 Dijkstra 算法的速度相当，唯一变得更差的是懒惰 Dijkstra 算法的优先队列占用了更多的空间。

#include "lazy-dijkstra.h"

int main() {
  const char *names = "stxyz";
  Graph G(5);
  G.add_edge(0, 1, 10);
  G.add_edge(0, 3, 5);
  G.add_edge(1, 2, 1);
  G.add_edge(1, 3, 2);
  G.add_edge(2, 4, 4);
  G.add_edge(3, 1, 3);
  G.add_edge(3, 2, 9);
  G.add_edge(3, 4, 2);
  G.add_edge(4, 0, 7);
  G.add_edge(4, 2, 6);

  LazyDijkstra LD;
  LD.lazy_dijkstra(G, 0);
  std::cout << "从源点 " << names[0] << " 出发的最短路径：" << std::endl;
  for (int v = 0; v < G.n; v++) {
    std::vector<int> p = LD.reconstruct_shortest_path(v);
    std::cout << names[v] << " " << LD.dist[v] << " ";
    for (int i = 0; i < p.size(); i++) {
      std::cout << names[p[i]] << (i < p.size() - 1 ? "->" : "");
    }
    std::cout << std::endl;
  }

  return 0;
}

#ifndef LAZY_DIJKSTRA_H
#define LAZY_DIJKSTRA_H

#include "graph.h"
#include "priority-queue.h"

class LazyDijkstra {
public:
  std::vector<double> dist;
  std::vector<int> prev;

  bool relax(int u, int v, double w) {
    if (dist[v] > dist[u] + w) {
      dist[v] = dist[u] + w;
      prev[v] = u;
      return true;
    } else {
      return false;
    }
  }

  void lazy_dijkstra(const Graph &G, int s) {
    dist.assign(G.n, inf);
    prev.assign(G.n, nil);
    dist[s] = 0;
    PriorityQueue Q;
    Q.insert(s, 0);
    while (not Q.empty()) {
      auto [u, du] = Q.extract_min();
      if (dist[u] == du) {
        for (auto [v, w] : G.A[u]) {
          if (relax(u, v, w)) {
            Q.insert(v, dist[v]);
          }
        }
      }
    }
  }

  std::vector<int> reconstruct_shortest_path(int v) {
    if (dist[v] == inf) {
      return {};
    } else {
      std::vector<int> p;
      for (int x = v; x != nil; x = prev[x]) {
        p.push_back(x);
      }
      std::reverse(p.begin(), p.end());
      return p;
    }
  }
};

#endif

#ifndef PRIORITY_QUEUE_H
#define PRIORITY_QUEUE_H

#include <bits/stdc++.h>

class PriorityQueue {
private:
  std::priority_queue<std::pair<int, double>,
                      std::vector<std::pair<int, double>>,
                      std::function<bool(std::pair<int, double> &,
                                         std::pair<int, double> &)>> data;

public:
  PriorityQueue() : data([](auto &a, auto &b) { return a.second > b.second; }) {}

  bool empty() { return data.empty(); }

  std::pair<int, double> extract_min() {
    std::pair<int, double> min = data.top();
    data.pop();
    return min;
  }

  void insert(int v, double d) { data.emplace(v, d); }
};

#endif

#ifndef GRAPH_H
#define GRAPH_H

#include <bits/stdc++.h>

class Graph {
public:
  int n;
  std::vector<std::tuple<int, int, double>> E;
  std::vector<std::vector<std::pair<int, double>>> A;

  Graph(int n) : n(n), E(), A(n) {}

  void add_edge(int u, int v, double w) {
    E.emplace_back(u, v, w);
    A[u].emplace_back(v, w);
  }

  void clear() {
    E.clear();
    A.assign(n, {});
  }
};

const double inf = std::numeric_limits<double>::infinity();
const int nil = -1;

#endif
